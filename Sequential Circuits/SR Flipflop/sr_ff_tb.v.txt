`timescale 1ns/1ps

module sr_ff_tb;

reg s,r,clk;
wire q;

SR_ff dut(s,r,clk,q);

initial begin
   $dumpfile("dump.vcd");
  $dumpvars(0, sr_ff_tb);
end

always #5 clk = ~clk;

initial
begin
    clk = 0;

    s = 0; r = 0; #10;
    s = 1; r = 0; #10;
    s = 0; r = 1; #10;
    s = 1; r = 1; #10;

    $finish;
end

initial
begin
  $monitor("time=%0t clk=%b s=%b r=%b q=%b",
              $time, clk, s, r, q);
end

endmodule
